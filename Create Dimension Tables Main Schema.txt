create or replace TABLE COVID_DATA.COVID_DATASET_2019.COUNTRY_DIM (
	COUNTRY_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 order,
	COUNTRY_NAME VARCHAR(255),
    CREATED_DATE TIMESTAMP DEFAULT CURRENT_DATE(),
    CREATED_BY VARCHAR(50) DEFAULT CURRENT_USER(),
	primary key (COUNTRY_ID)
);

create or replace TABLE COVID_DATA.COVID_DATASET_2019.STATE_DIM (
	STATE_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 order,
	-- COUNTRY_ID NUMBER(38,0),
	REGION_NAME VARCHAR(255),
	LAT NUMERIC(38,9),
	LON NUMERIC(38,9),
    CREATED_DATE TIMESTAMP DEFAULT CURRENT_DATE(),
    CREATED_BY VARCHAR(50) DEFAULT CURRENT_USER(),
	primary key (STATE_ID)
	-- foreign key (COUNTRY_ID) references COVID_DATA.COVID_DATASET_2019.COUNTRY_DIM(COUNTRY_ID)
);

create or replace TABLE COVID_DATA.COVID_DATASET_2019.WHO_DIM (
	WREGION_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 order,
	REGION_NAME VARCHAR(255),
    CREATED_DATE TIMESTAMP DEFAULT CURRENT_DATE(),
    CREATED_BY VARCHAR(50) DEFAULT CURRENT_USER(),
	primary key (WREGION_ID)
);

create or replace TABLE COVID_DATA.COVID_DATASET_2019.DATE_DIM (
	DATE_ID NUMBER(38,0) NOT NULL autoincrement start 1 increment 1 order,
	DATE DATE,
	MONTH VARCHAR(255),
	DAY VARCHAR(255),
	YEAR NUMBER(38,0),
	MONTH_NUMBER NUMBER(38,0),
	DAY_NUMBER NUMBER(38,0),
	primary key (DATE_ID)
);
